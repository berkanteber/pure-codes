times = { "noopt": {}, "sexeq": {}, "egal": {}, "minreg": {}, "maxcar": {}}
for i in range(int(len(lines) / 7)):
    name = lines[7 * i].split(' ')[2][6:-10]
    
    times["noopt"][name] = float(lines[7 * i + 1].split(' ')[-2])
    times["sexeq"][name] = float(lines[7 * i + 2].split(' ')[-2])
    times["egal"][name] = float(lines[7 * i + 3].split(' ')[-2])
    times["minreg"][name] = float(lines[7 * i + 4].split(' ')[-2])
    times["maxcar"][name] = float(lines[7 * i + 5].split(' ')[-2])
